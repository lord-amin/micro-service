درصورتی که ویژگی های زیر در فایل پیکر بندی وجود داشته باشند

server.port=1998
server.address=192.168.100.185
server.contextPath=/auth-server/v5
springfox.documentation.swagger.v2.path=/api-docs
springfox.documentation.swagger.v1.path=/v1/api-docs

http://192.168.100.185:1998/auth-server/v5
